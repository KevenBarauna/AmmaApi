CREATE TABLE usuario
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[nome] VARCHAR(40) NOT NULL,
[senha] VARCHAR(30) NOT NULL,
[email] VARCHAR(35),
[cargo] VARCHAR(35),
[codAvatar] INT NOT NULL,
[idPermissao] INT NOT NULL,
FOREIGN KEY (idPermissao) REFERENCES permissao(id)
)

CREATE TABLE permissao
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[descricao] VARCHAR(15) NOT NULL, 
)

CREATE TABLE categoria
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[descricao] VARCHAR(15) NOT NULL, 
)

CREATE TABLE status
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[valor] VARCHAR(25) NOT NULL, 
[descricao] VARCHAR(90) NOT NULL, 
)

CREATE TABLE sugestao
(
[id] INT NOT NULL PRIMARY KEY IDENTITY, 
[idUsuario] INT NOT NULL,
[idCategoria] INT NOT NULL,
[idStatus] INT NOT NULL,
[isAnonimo] BIT NOT NULL,
[titulo] VARCHAR(30) NOT NULL, 
[descricao] VARCHAR(200) NOT NULL, 
[quantidadeVotosPositivos] INT NOT NULL,
[quantidadeVotosNegativos] INT NOT NULL,
[dataSugestao] datetime NOT NULL,
FOREIGN KEY (idUsuario) REFERENCES usuario(id),
FOREIGN KEY (idCategoria) REFERENCES categoria(id),
FOREIGN KEY (idStatus) REFERENCES status(id)
)
